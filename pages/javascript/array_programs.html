Find Duplicate Elements 

        // Function to find duplicate elements in an array
        function findDuplicates(arr) {
            const counts = {};
            const duplicates = [];
            arr.forEach(num => {
                counts[num] = (counts[num] || 0) + 1;
            });
            for (const [key, value] of Object.entries(counts)) {
                if (value > 1) {
                    duplicates.push(Number(key));
                }
            }
            return duplicates;
        }
        
        // Example usage
        const arr = [1, 2, 3, 2, 4, 5, 6, 5];
        console.log(findDuplicates(arr)); // Output: [2, 5]
Reverse an Array Using Loop 

        // Function to reverse an array using a loop
        function reverseArray(arr) {
            let start = 0;
            let end = arr.length - 1;
        
            while (start < end) {
                // Swap elements at 'start' and 'end'
                let temp = arr[start];
                arr[start] = arr[end];
                arr[end] = temp;
        
                // Move 'start' forward and 'end' backward
                start++;
                end--;
            }
        
            return arr;
        }
        
        // Example usage
        const arr = [1, 2, 3, 4, 5];
        console.log(reverseArray(arr)); // Output: [5, 4, 3, 2, 1]

Find Missing Number in a Sequence 

        // Function to find the missing number in a sequence
        function findMissingNumber(arr, n) {
            const totalSum = (n * (n + 1)) / 2;
            const arraySum = arr.reduce((sum, num) => sum + num, 0);
            return totalSum - arraySum;
        }
        
        // Example usage
        const arr = [1, 2, 4, 5, 6];
        console.log(findMissingNumber(arr, 6)); // Output: 3
Array Rotation 

        // Function to rotate an array by d positions
        function rotateArray(arr, d) {
            d = d % arr.length;
            return arr.slice(d).concat(arr.slice(0, d));
        }
        
        // Example usage
        const arr = [1, 2, 3, 4, 5];
        console.log(rotateArray(arr, 2)); // Output: [3, 4, 5, 1, 2]

        // Function to find pairs with a target sum
        function findPairs(arr, target) {
            const seen = new Set();
            const pairs = [];
            for (const num of arr) {
                const complement = target - num;
                if (seen.has(complement)) {
                    pairs.push([complement, num]);
                }
                seen.add(num);
            }
            return pairs;
        }
        
        // Example usage
        const arr = [1, 2, 3, 4, 5];
        console.log(findPairs(arr, 6)); // Output: [[2, 4], [1, 5]]

Merge Two Sorted Arrays 

        // Function to merge two sorted arrays
        function mergeSortedArrays(arr1, arr2) {
            let i = 0, j = 0, merged = [];
            while (i < arr1.length && j < arr2.length) {
                if (arr1[i] < arr2[j]) {
                    merged.push(arr1[i++]);
                } else {
                    merged.push(arr2[j++]);
                }
            }
            return merged.concat(arr1.slice(i)).concat(arr2.slice(j));
        }
        
        // Example usage
        const arr1 = [1, 3, 5];
        const arr2 = [2, 4, 6];
        console.log(mergeSortedArrays(arr1, arr2)); // Output: [1, 2, 3, 4, 5, 6]
Find Intersection of Two Arrays 

        // Function to find intersection of two arrays
        function findIntersection(arr1, arr2) {
            const set1 = new Set(arr1);
            const set2 = new Set(arr2);
            return [...set1].filter(num => set2.has(num));
        }
        
        // Example usage
        const arr1 = [1, 2, 3, 4];
        const arr2 = [3, 4, 5, 6];
        console.log(findIntersection(arr1, arr2)); // Output: [3, 4]

Find All Unique Triplets That Sum to Zero 

        // Function to find all unique triplets that sum to zero
        function findTriplets(arr) {
            const result = [];
            arr.sort((a, b) => a - b);
            for (let i = 0; i < arr.length - 2; i++) {
                if (i > 0 && arr[i] === arr[i - 1]) continue; // Skip duplicates
                let left = i + 1, right = arr.length - 1;
                while (left < right) {
                    const sum = arr[i] + arr[left] + arr[right];
                    if (sum === 0) {
                        result.push([arr[i], arr[left], arr[right]]);
                        while (arr[left] === arr[left + 1]) left++;
                        while (arr[right] === arr[right - 1]) right--;
                        left++;
                        right--;
                    } else if (sum < 0) {
                        left++;
                    } else {
                        right--;
                    }
                }
            }
            return result;
        }
        
        // Example usage
        const arr = [-1, 0, 1, 2, -1, -4];
        console.log(findTriplets(arr)); // Output: [[-1, -1, 2], [-1, 0, 1]]

        // Function to find the majority element in an array
        function findMajorityElement(arr) {
            const counts = {};
            for (const num of arr) {
                counts[num] = (counts[num] || 0) + 1;
                if (counts[num] > arr.length / 2) {
                    return num;
                }
            }
            return -1;
        }

Find Maximum Subarray Sum (Kadane's Algorithm) 

        // Function to find the maximum subarray sum using Kadane's Algorithm
        function maxSubArraySum(arr) {
            let maxSum = arr[0];
            let currentSum = arr[0];
            for (let i = 1; i < arr.length; i++) {
                currentSum = Math.max(arr[i], currentSum + arr[i]);
                maxSum = Math.max(maxSum, currentSum);
            }
            return maxSum;
        }
        
        // Example usage
        const arr = [-2, 1, -3, 4, -1, 2, 1, -5, 4];
        console.log(maxSubArraySum(arr)); // Output: 6 (subarray [4, -1, 2, 1])
Check if Array Contains a Subsequence 

        // Function to check if an array contains a subsequence
        function containsSubsequence(arr, subsequence) {
            let i = 0, j = 0;
            while (i < arr.length && j < subsequence.length) {
                if (arr[i] === subsequence[j]) {
                    j++;
                }
                i++;
            }
            return j === subsequence.length;
        }
        
        // Example usage
        const arr = [1, 2, 3, 4, 5];
        const subsequence = [2, 4];
        console.log(containsSubsequence(arr, subsequence)); // Output: true
Move All Zeros to End 

        // Function to move all zeros to the end of an array
        function moveZerosToEnd(arr) {
            let nonZeroIndex = 0;
            for (let i = 0; i < arr.length; i++) {
                if (arr[i] !== 0) {
                    arr[nonZeroIndex++] = arr[i];
                }
            }
            while (nonZeroIndex < arr.length) {
                arr[nonZeroIndex++] = 0;
            }
            return arr;
        }
        
        // Example usage
        const arr = [0, 1, 0, 3, 12];
        console.log(moveZerosToEnd(arr)); // Output: [1, 3, 12, 0, 0]

        // Function to check if an array is a palindrome
        function isPalindrome(arr) {
            let start = 0, end = arr.length - 1;
            while (start < end) {
                if (arr[start] !== arr[end]) {
                    return false;
                }
                start++;
                end--;
            }
            return true;
        }
        
        // Example usage
        const arr = [1, 2, 3, 2, 1];
        console.log(isPalindrome(arr)); // Output: true

        // Function to find a subarray with a given sum
        function subarrayWithGivenSum(arr, targetSum) {
            let sum = 0;
            let start = 0;
            for (let end = 0; end < arr.length; end++) {
                sum += arr[end];
                while (sum > targetSum && start < end) {
                    sum -= arr[start++];
                }
                if (sum === targetSum) {
                    return arr.slice(start, end + 1);
                }
            }
            return null;
        }
        
        // Example usage
        const arr = [1, 2, 3, 7, 5];
        console.log(subarrayWithGivenSum(arr, 12)); // Output: [2, 3, 7]
Binary Search on Sorted Array 

        // Function to perform binary search on a sorted array
        function binarySearch(arr, target) {
            let left = 0, right = arr.length - 1;
            while (left <= right) {
                const mid = Math.floor((left + right) / 2);
                if (arr[mid] === target) {
                    return mid;
                }
                if (arr[mid] < target) {
                    left = mid + 1;
                } else {
                    right = mid - 1;
                }
            }
            return -1;
        }
        
        // Example usage
        const arr = [1, 3, 5, 7, 9];
        console.log(binarySearch(arr, 5)); // Output: 2
        console.log(binarySearch(arr, 4)); // Output: -1
Find All Unique Combinations that Sum to Target 

        // Function to find all unique combinations that sum to a target
        function findCombinations(arr, target) {
            const result = [];
            function backtrack(start, target, current) {
                if (target === 0) {
                    result.push([...current]);
                    return;
                }
                for (let i = start; i < arr.length; i++) {
                    if (i > start && arr[i] === arr[i - 1]) continue; // Skip duplicates
                    if (arr[i] > target) break; // No need to continue if the number exceeds the target
                    current.push(arr[i]);
                    backtrack(i + 1, target - arr[i], current);
                    current.pop();
                }
            }
            arr.sort((a, b) => a - b); // Sorting helps in skipping duplicates
            backtrack(0, target, []);
            return result;
        }
        
        // Example usage
        const arr = [10, 1, 2, 7, 6, 5];
        console.log(findCombinations(arr, 8)); // Output: [[1, 2, 5], [1, 7], [2, 6]]

Find the Longest Consecutive Sequence 

        // Function to find the longest consecutive sequence in an array
        function longestConsecutiveSequence(arr) {
            const set = new Set(arr);
            let longest = 0;
            for (const num of arr) {
                if (!set.has(num - 1)) {
                    let currentNum = num;
                    let currentStreak = 1;
                    while (set.has(currentNum + 1)) {
                        currentNum++;
                        currentStreak++;
                    }
                    longest = Math.max(longest, currentStreak);
                }
            }
            return longest;
        }
        
        // Example usage
        const arr = [100, 4, 200, 1, 3, 2];
        console.log(longestConsecutiveSequence(arr)); // Output: 4 (sequence: [1, 2, 3, 4])

        // Function to check if an array can be split into two equal sum parts
        function canSplitArray(arr) {
            const totalSum = arr.reduce((sum, num) => sum + num, 0);
            if (totalSum % 2 !== 0) return false; // If total sum is odd, it's not possible to split equally
            const target = totalSum / 2;
            const dp = new Set([0]);
            for (const num of arr) {
                const nextSet = new Set(dp);
                for (const sum of dp) {
                    nextSet.add(sum + num);
                }
                dp.clear();
                for (const newSum of nextSet) {
                    dp.add(newSum);
                    if (newSum === target) return true;
                }
            }
            return false;
        }
        
        // Example usage
        const arr = [1, 5, 11, 5];
        console.log(canSplitArray(arr)); // Output: true
Find Peak Element in an Array 

        // Function to find a peak element in an array
        function findPeak(arr) {
            let left = 0, right = arr.length - 1;
            while (left <= right) {
                const mid = Math.floor((left + right) / 2);
                if ((mid === 0 || arr[mid - 1] <= arr[mid]) && (mid === arr.length - 1 || arr[mid + 1] <= arr[mid])) {
                    return mid;
                }
                if (mid > 0 && arr[mid - 1] > arr[mid]) {
                    right = mid - 1;
                } else {
                    left = mid + 1;
                }
            }
        }
        
        // Example usage
        const arr = [1, 3, 20, 4, 1];
        console.log(findPeak(arr)); // Output: 2 (index of 20)
Rearrange Array Alternating Positive and Negative 

        // Function to rearrange an array such that positive and negative numbers alternate
        function rearrangeAlternating(arr) {
            const pos = arr.filter(x => x >= 0);
            const neg = arr.filter(x => x < 0);
            const result = [];
            let i = 0, j = 0;
            while (i < pos.length && j < neg.length) {
                result.push(pos[i++]);
                result.push(neg[j++]);
            }
            while (i < pos.length) result.push(pos[i++]);
            while (j < neg.length) result.push(neg[j++]);
            return result;
        }
        
        // Example usage
        const arr = [1, -2, 3, -4, 5, -6];
        console.log(rearrangeAlternating(arr)); // Output: [1, -2, 3, -4, 5, -6]

        // Function to check if an array is sorted
        function isArraySorted(arr) {
            for (let i = 0; i < arr.length - 1; i++) {
                if (arr[i] > arr[i + 1]) {

                }

Find Elements that Occur More than n/k Times 

        // Function to find elements that occur more than n/k times
        function findElementsMoreThanK(arr, k) {
            const result = [];
            const n = arr.length;
            const freqMap = {};
            for (const num of arr) {
                freqMap[num] = (freqMap[num] || 0) + 1;
            }
            for (const [num, count] of Object.entries(freqMap)) {
                if (count > n / k) {
                    result.push(num);
                }
            }
            return result;
        }
        
        // Example usage
        const arr = [1, 2, 3, 1, 2, 1, 1, 3, 2];
        console.log(findElementsMoreThanK(arr, 3)); // Output: [1, 2]
