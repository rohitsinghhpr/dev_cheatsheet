Reverse a String Using Loop 

        // Function to reverse a string using a loop
        function reverseString(str) {
            let reversedStr = '';
            for (let i = str.length - 1; i >= 0; i--) {
                reversedStr += str[i];
            }
            return reversedStr;
        }
        
        // Example usage
        const str = "hello world";
        console.log(reverseString(str)); // Output: "dlrow olleh"

            return str.split('').reverse().join('');
        }
        
        // Example usage

        console.log(reverseString(str)); // Output: "olleh"
Check if a string is a palindrome 

        // Function to check if a string is a palindrome
        function isPalindrome(str) {
            const reversed = str.split('').reverse().join('');
            return str === reversed;
        }
        
        // Example usage
        const str = "madam";
        console.log(isPalindrome(str)); // Output: true
Count vowels and consonants in a string 

        // Function to count vowels and consonants in a string
        function countVowelsAndConsonants(str) {
            let vowels = 0, consonants = 0;
            for (let char of str.toLowerCase()) {
                if ('aeiou'.includes(char)) vowels++;
                else if (/[a-z]/.test(char)) consonants++;
            }
            return { vowels, consonants };
        }
        
        // Example usage
        const str = "hello world";
        console.log(countVowelsAndConsonants(str)); // Output: { vowels: 3, consonants: 7 }
Find the frequency of characters in a string 

        // Function to find the frequency of characters in a string
        function charFrequency(str) {
            const freqMap = {};
            for (let char of str) {
                freqMap[char] = (freqMap[char] || 0) + 1;
            }
            return freqMap;
        }
        
        // Example usage
        const str = "hello";
        console.log(charFrequency(str)); // Output: { h: 1, e: 1, l: 2, o: 1 }
Check if two strings are anagrams 

        // Function to check if two strings are anagrams
        function areAnagrams(str1, str2) {
            const sortedStr1 = str1.split('').sort().join('');
            const sortedStr2 = str2.split('').sort().join('');
            return sortedStr1 === sortedStr2;
        }
        
        // Example usage
        const str1 = "listen";
        const str2 = "silent";
        console.log(areAnagrams(str1, str2)); // Output: true
Find all permutations of a string 

        // Function to find all permutations of a string
        function permute(str) {
            const results = [];
            if (str.length === 1) return [str];
            for (let i = 0; i < str.length; i++) {
                const char = str[i];
                const remainingChars = str.slice(0, i) + str.slice(i + 1);
                const remainingPermutations = permute(remainingChars);
                for (const perm of remainingPermutations) {
                    results.push(char + perm);
                }
            }
            return results;
        }
        
        // Example usage
        const str = "abc";
        console.log(permute(str)); // Output: ["abc", "acb", "bac", "bca", "cab", "cba"]

Find the longest word in a sentence 

        // Function to find the longest word in a sentence
        function longestWord(sentence) {
            const words = sentence.split(' ');
            let longest = '';
            for (let word of words) {
                if (word.length > longest.length) longest = word;
            }
            return longest;
        }
        
        // Example usage
        const sentence = "The quick brown fox jumps over the lazy dog";
        console.log(longestWord(sentence)); // Output: "jumps"

Convert a string to title case 

        // Function to convert a string to title case
        function toTitleCase(str) {
            return str.split(' ').map(word => word.charAt(0).toUpperCase() + word.slice(1).toLowerCase()).join(' ');

Find all substrings of a string 

        // Function to find all substrings of a string
        function getAllSubstrings(str) {
            const substrings = [];
            for (let i = 0; i < str.length; i++) {
                for (let j = i + 1; j <= str.length; j++) {
                    substrings.push(str.slice(i, j));
                }
            }
            return substrings;
        }
        
        // Example usage
        const str = "abc";
        console.log(getAllSubstrings(str)); // Output: ["a", "ab", "abc", "b", "bc", "c"]

        // Function to check if a string is a valid palindrome ignoring case and non-alphanumeric characters
        function isValidPalindrome(str) {
            const cleanedStr = str.replace(/[^a-zA-Z0-9]/g, '').toLowerCase();
            const reversedStr = cleanedStr.split('').reverse().join('');
            return cleanedStr === reversedStr;
        }
        
        // Example usage
        const str = "A man, a plan, a canal, Panama!";
        console.log(isValidPalindrome(str)); // Output: true

        function firstNonRepeating(str) {
            const freqMap = {};
            for (let char of str) {
                freqMap[char] = (freqMap[char] || 0) + 1;

Find the longest substring without repeating characters 

        // Function to find the longest substring without repeating characters
        function longestSubstring(str) {
            let map = new Map();
            let start = 0, maxLength = 0;
            
            for (let end = 0; end < str.length; end++) {
                if (map.has(str[end])) {
                    start = Math.max(map.get(str[end]) + 1, start);
                }
                map.set(str[end], end);
                maxLength = Math.max(maxLength, end - start + 1);
            }
            return maxLength;
        }
        
        // Example usage
        const str = "abcabcbb";
        console.log(longestSubstring(str)); // Output: 3 (The longest substring is "abc")

Check if a string is a rotation of another string 

        // Function to check if a string is a rotation of another string
        function isRotation(str1, str2) {
            if (str1.length !== str2.length) return false;
            return (str1 + str1).includes(str2);
        }
        
        // Example usage
        const str1 = "abcde";
        const str2 = "deabc";
        console.log(isRotation(str1, str2)); // Output: true

Find the length of the longest palindromic substring 

        // Function to find the length of the longest palindromic substring
        function longestPalindromicSubstring(str) {
            let maxLength = 1;
        
            function expandAroundCenter(left, right) {
                while (left >= 0 && right < str.length && str[left] === str[right]) {
                    maxLength = Math.max(maxLength, right - left + 1);
                    left--;
                    right++;
                }
            }
        
            for (let i = 0; i < str.length; i++) {
                expandAroundCenter(i, i); // Odd length palindrome
                expandAroundCenter(i, i + 1); // Even length palindrome
            }
        
            return maxLength;
        }
        
        // Example usage
        const str = "babad";
        console.log(longestPalindromicSubstring(str)); // Output: 3 (The longest palindrome is "aba")
Check if a string has all unique characters 

        // Function to check if a string has all unique characters
        function hasUniqueCharacters(str) {
            const charSet = new Set();
            for (let char of str) {
                if (charSet.has(char)) return false;
                charSet.add(char);
            }
            return true;
        }
        
        // Example usage
        const str = "abcdef";
        console.log(hasUniqueCharacters(str)); // Output: true
        const str2 = "aabbcc";
        console.log(hasUniqueCharacters(str2)); // Output: false
